\subsection{Parameter Types}

\begin{asparadesc}
\item[\typeitem{SIZE}]
A number specifying the size. Needs to be greater then zero, for
example \codequoted{5}.
\item[\typeitem{FILE}]
A path to a file. The file should be exists and be readable, for example
\codequoted{./file}.
\item[\typeitem{LIST}]
Items separated with either a comma ``,'' or a semicolumn ``;'', for example
\codequoted{0,1,2,3} or \codequoted{0;1;2;3}. The separator can be mixed.
\item[\typeitem{NAME}]
A name identifier. For example \codequoted{name}.
\item[\typeitem{CHAR}]
A single character. For example \codequoted{,}.
\item[\typeitem{ENUM}]
One of the valid identifiers. For example \codequoted{NAME}.
\end{asparadesc}
