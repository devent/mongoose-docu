\section{Types}

\subsection{Return Values}

%%
\label{par:returnvalue}
\parparameter{\parcommand{ReturnValue}}

The return value from a command. Returns a human readable message
for this return value.

%%
\label{par:returnvalue}
\parparameter{\parcommand{ReturnCode}}

The numerical return code of a command.

\subparagraph{Properties}

\begin{asparadesc}
%
\item[asNumber] Returns the number value of the return value.
%
\item[asBoolean] Returns the boolean value of the return value. As a convention the return
value \code{0} equals \code{true} and any value not \code{0} equals
\code{false}.
%
\end{asparadesc}

\subparagraph{Comparison}

\begin{compactitem}
%
\item to \code{Boolean}: if this value as boolean equals to the specified boolean
value.
%
\item to \code{Number}: if this value as number equals to the specified number value.
%
\item to \code{ReturnCode}: if this value as number equals to the specified return code
number value.
%
\end{compactitem}

%%
\label{par:fileslist}
\parparameter{\parcommand{FilesList}}

Returns found files from the \parcommand{listFiles} build-in.

\subparagraph{Properties}

\begin{asparadesc}
%
\item[files] Returns the found files.
%
\end{asparadesc}

\subparagraph{Comparison}

\begin{compactitem}
%
\item to \code{List}: if this found files equals to the files in the specified
list.
%
\item to \code{FilesList}: if this found files equals to the files in the specified
return value.
%
\end{compactitem}

%%
\label{par:parametervalue}
\parparameter{\parcommand{ParameterValue}}

Returns the parsed parameters from the build-in command \parcommand{parse}.
If the parameter are valid, then the properties of the parameter bean class
can be accessed.

\subparagraph{Methods}

\begin{asparadesc}
%
\item[printExample()] Print an example of the command line arguments in the
standard output.
%
\item[printSingleLineUsage()] Print an example of the usage of the command
line arguments in one line in the standard output.
%
\item[printUsage()] Print the usage of the command line arguments in one line in the standard
output.
%
\item[valid(Closure)] Runs the specified closure if the parameter are valid.
%
\item[notValid(Closure)] Runs the specified closure if the parameter are not valid.
%
\end{asparadesc}

\subparagraph{Properties}

\begin{asparadesc}
%
\item[example] Returns an example of the command line arguments.
%
\item[singleLineUsage] Returns an usage example in a single line of the
command line arguments.
%
\item[usage] Returns an usage example of the command line arguments.
%
\item[isValid] Returns whether or not the arguments are parsed and the bean class
contains the valid parameter.
%
\end{asparadesc}

\subparagraph{Comparison}

\begin{compactitem}
%
\item to \code{Boolean}: if this value valid flag equals the specified boolean value.
%
\item to \code{ParameterValue}: if this value valid flag equals the specified value valid
flag and the parameter bean object is the same.
%
\end{compactitem}


%%
\label{par:filevalue}
\parparameter{\parcommand{FileValue}}

Value for a file name.

\subparagraph{Methods}

\begin{asparadesc}
%
\item[asType(Class)] Returns this value as the specified type.
The type can be of class:
\begin{compactitem}
\item \code{String}: returns the absolute path of the file.
\item \code{File}: returns the file.
\end{compactitem}
%
\end{asparadesc}

\subparagraph{Properties}

\begin{asparadesc}
%
\item[file] Returns the file.
%
\end{asparadesc}

\subparagraph{Comparison}

\begin{compactitem}
%
\item to \code{Object}: if this file equals to the string representation of the
object interpreted as a file path.
%
\item to \code{File}: if this file equals to the specified file.
%
\item to \code{FileValue}: if this file equals to the specified value file.
%
\end{compactitem}

\subparagraph{Used By}

\begin{compactitem}
%
\item \parcommand{detox}, see \refpar{par:detox};
%
\end{compactitem}
